git init

git add имя файла

git rm --cached имя файла * - все файлы

git commit -m "комментарий"

git branch имя - создать ветку 

git checkout имя - переключиться на ветку, вернуться на ветку
git checkout id коммита - переключиться на определенный коммит

git status, git log 

git push
git push --set-upstream origin 123  - создание удаленной ветки

файл .gitignore - исключения для коммитов index.html  css/  js/*.js

git log --oneline

git revert номер коммита - отменить коммит, для выхода :wq

git reset номер - удалить все коммиты до определенного, при этом остаются изменения в файлах
git reset номер --hard - жесткое удаление коммита с откатом изменений

git branch имя - создание ветки (ответвления)
git checkout имя - переход на ветку

git branch -a - отобразить все ветки
git checkout -b имя ветки - создание ветки и переход на нее

git merge имя ветки с которым объединяется

git remote add origin адрес - подключение удаленного репозитория
git remote - показать к какому удаленному подключен
git push -u origin (имя репозитория) master(имя ветки которую пушим) - добавить файлы в удаленный репозиторий, -u (--set-upstream) - устанавливает удаленную ветку по умолчанию для текущей локальной ветки

git clone адрес - клонирование репозитория в локальную папку

git pull
